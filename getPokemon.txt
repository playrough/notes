
//name
const name = document.querySelector('.sv-tabs-tab.active').innerText;
const id = document.querySelector('.sv-tabs-tab.active');

const data = document.querySelectorAll(`#${id.href.split('#')[1]} .vitals-table`);
const rows = data[0].querySelectorAll('tr');

//number
const number = rows[0].querySelector('td strong').innerText;

//type
const type = [];
rows[1].querySelectorAll('td a').forEach(a=>{
    type.push(a.innerText.charAt(0) + a.innerText.toLowerCase().slice(1));
});


//abilities
const abilityTD = rows[5].querySelector('td');
const abilityA = abilityTD.querySelectorAll('span a');
const abilitySmall = abilityTD.querySelector('small a') || null;
const abilities = {
			"ability1": abilityA[0].innerText,
			"ability2": abilityA[1]?.innerText || null,
			"hidden": abilitySmall?.innerText || null
};

const rows2 = data[3].querySelectorAll('tr');

const stats= [];
rows2.forEach(tr=>{
    const cells = tr.querySelectorAll('td');
    stats.push(Number(cells[0].innerText));
});

const hp = stats[0];
const attack = stats[1];
const defense = stats[2];
const spAttack = stats[3];
const spDefense = stats[4];
const speed = stats[5];
const total = stats[6];

const pokemon = {
    [name] : {
        number, name, type, abilities, hp, attack, defense, spAttack, spDefense, speed, total
    }
}

console.log(pokemon);
copy(pokemon);
